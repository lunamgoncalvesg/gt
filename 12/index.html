SQL de los pokemones

drop database if exists pokemon;
create database pokemon;
use pokemon;

create table pokemones(
id int auto_increment,
nom varchar(100) not null,
capt date,
pokerus boolean,
constraint pk_id primary key (id)
);

insert into pokemones (nom, capt, pokerus) values
('pikachu', '2000-06-02', 1),
('articuno', '1998-04-20', 0),
('squirtle', '2003-02-01', 0),
('mew', '2004-07-23', 1),
('jynx', '1997-01-30', 1);

select * from pokemones;
select nom from pokemones where nom like '%i%';
select nom, capt from pokemones order by capt;
